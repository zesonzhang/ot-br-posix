199template <typename T> class Optional
200{
201public:
202 constexpr Optional(void) = default;
203
204 Optional(T aValue) { SetValue(aValue); }
205
206 ~Optional(void) { ClearValue(); }
207
208 Optional(const Optional &aOther) { AssignFrom(aOther); }
209
210 Optional &operator=(const Optional &aOther) { AssignFrom(aOther); }
211
212 constexpr const T *operator->(void) const { return &GetValue(); }
213
214 constexpr const T &operator*(void) const { return GetValue(); }
215
216 constexpr bool HasValue(void) const { return mHasValue; }
217
218private:
219 T &GetValue(void) const
220 {
221 assert(mHasValue);
222 return *const_cast<T *>(reinterpret_cast<const T *>(&mStorage));
223 }
224
225 void ClearValue(void)
226 {
227 if (mHasValue)
228 {
229 GetValue().~T();
230 mHasValue = false;
231 }
232 }
233
234 void SetValue(const T &aValue)
235 {
236 ClearValue();
237 new (&mStorage) T(aValue);
238 mHasValue = true;
239 }
240
241 void AssignFrom(const Optional &aOther)
242 {
243 ClearValue();
244 if (aOther.mHasValue)
245 {
246 SetValue(aOther.GetValue());
247 }
248 }
249
250 alignas(T) unsigned char mStorage[sizeof(T)];
251 bool mHasValue = false;
252};
